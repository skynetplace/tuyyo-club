## 1. Неконсистентність спінерів у додатку

### Опис проблеми

Наразі існують дві окремі реалізації спінерів завантаження:
- **Головна сторінка** — спинер відображається в секціях з HTMX-запитами (inventory-list, locations-list, reviews-list)
- **Сторінка бронювання** — спинер у блоці `booking-wizard` при завантаженні кроку 1
- **Адмін-панель** — при переході на будь-яку секцію окрім Dashboard та Settings відображається тільки спинер (нескінченний стан завантаження) без рендерингу контенту

### Технічні деталі

**Файли що задіяні:**
- `web/public/index.html` — секції з `hx-trigger="load"`
- `web/public/booking.html` — блок `booking-wizard` з `hx-get="/api/booking/step-1"`
- `web/public/admin/index.html` — всі секції окрім dashboard та settings

**Кореневі причини:**

1. **Відсутність обробки помилок в HTMX:**
   ```javascript
   // Поточний код — немає обробки помилок
   <div hx-get="https://tuyyo-api.sashka-desire.workers.dev/api/inventory"
        hx-trigger="load"
        hx-swap="innerHTML">
     <div class="text-center"><div class="spinner-border text-primary"></div></div>
   </div>
   ```
   Якщо запит не вдається (CORS, мережа, 500), HTMX не має механізму для приховування спінера.

2. **Неправильне управління станом завантаження:**
   - Спинер показується за замовчуванням (`display: block`)
   - HTMX не приховує спинер після завершення запиту (успішного або ні)
   - Відсутній стан `hx-error` для відображення помилок

3. **Адмін-панель — проблема з асинхронними запитами:**
   ```javascript
   // admin/index.html — проблемний код
   async function loadSection(name) {
     // Відправляємо запит
     const stats = await api('/dashboard');
     // Але якщо запит падає — спинер залишається
     // Немає catch() блоку!
   }
   ```

4. **Race conditions:**
   - Кілька HTMX-запитась стартують одночасно
   - Немає координації між запитами
   - Один запит може завершитися з помилкою інший — успішно

// Глобальний обробник помилок HTMX
document.body.addEventListener('htmx:responseError', function(evt) {
  const container = evt.target.closest('.section-container');
  if (container) {
    container.classList.add('htmx-error');
    const errorMsg = container.querySelector('.error-message');
    if (errorMsg) {
      errorMsg.textContent = `Error ${evt.detail.xhr.status}: Failed to load data`;
    }
  }
});

// Функція для повторного завантаження
function retryLoad(btn) {
  const container = btn.closest('.section-container');
  container.classList.remove('htmx-error');
  htmx.trigger(container, 'load');
}
```

// admin/index.html — виправлений код
async function loadSection(name) {
  const content = document.getElementById('admin-content');

  // Показуємо спинер
  content.innerHTML = '<div class="text-center py-5"><div class="spinner-border"></div></div>';

  try {
    let html = '';
    switch(name) {
      case 'dashboard':
        html = await renderDashboard();
        break;
      case 'bookings':
        html = await renderBookings();
        break;
      case 'inventory':
        html = await renderInventory();
        break;
      case 'locations':
        html = await renderLocations();
        break;
      case 'documents':
        html = await renderDocuments();
        break;
      case 'translations':
        html = await renderTranslations();
        break;
      case 'settings':
        html = await renderSettings();
        break;
      default:
        html = '<div class="alert alert-warning">Unknown section</div>';
    }
    content.innerHTML = html;
  } catch (error) {
    console.error(`Error loading section ${name}:`, error);
    content.innerHTML = `
      <div class="alert alert-danger m-4">
        <i class="bi bi-exclamation-triangle"></i>
        Failed to load ${name}
        <button class="btn btn-sm btn-outline-danger ms-2" onclick="loadSection('${name}')">
          Retry
        </button>
      </div>
    `;
  }
}
```

    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          video.play();

          video.pause();

    }, { threshold: 0.1 });

    observer.observe(video);

**JavaScript (оптимізований):**

```javascript
class ParallaxController {
  constructor() {
    this.sections = [];
    this.ticking = false;
    this.init();
  }

  init() {
    // Знаходимо всі секції з паралаксом
    document.querySelectorAll('.parallax-section').forEach(section => {
      const bg = section.querySelector('.parallax-bg');
      if (!bg) return;

      const imageUrl = bg.dataset.parallaxImage;
      const speed = parseFloat(bg.dataset.parallaxSpeed) || 0.5;

      // Завантажуємо зображення
      const img = new Image();
      img.onload = () => {
        bg.style.backgroundImage = `url('${imageUrl}')`;
      };
      img.src = imageUrl;

      this.sections.push({
        section,
        bg,
        speed
      });
    });

    // Використовуємо Intersection Observer для оптимізації
    this.setupIntersectionObserver();

    // Слухаємо скрол
    window.addEventListener('scroll', () => this.onScroll(), { passive: true });
  }

  setupIntersectionObserver() {
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        const section = this.sections.find(s => s.section === entry.target);
        if (section) {
          section.isVisible = entry.isIntersecting;
        }
      });
    }, { threshold: 0.1 });

    this.sections.forEach(s => observer.observe(s.section));
  }

  onScroll() {
    if (!this.ticking) {
      requestAnimationFrame(() => {
        this.updateParallax();
        this.ticking = false;
      });
      this.ticking = true;
    }
  }

  updateParallax() {
    const scrollY = window.pageYOffset;

    this.sections.forEach(({ section, bg, speed, isVisible }) => {
      if (!isVisible) return;

      const rect = section.getBoundingClientRect();
      const sectionTop = rect.top + scrollY;
      const sectionHeight = rect.height;

      // Перевіряємо чи секція видима
      if (rect.bottom < 0 || rect.top > window.innerHeight) return;

      // Рахуємо зміщення
      const offset = (scrollY - sectionTop) * speed;

      // Застосовуємо transform
      bg.style.transform = `translateY(${offset}px)`;
    });
  }
}

// Ініціалізація
document.addEventListener('DOMContentLoaded', () => {
  new ParallaxController();
});
```

**Вимоги до продуктивності:**
- Використовуємо `requestAnimationFrame` для плавності
- `will-change: transform` для GPU-прискорення
- `Intersection Observer` для оптимізації (тільки видимі секції)
- `{ passive: true }` для scroll listener
- Мобільні пристрої: можна вимкнути паралакс для кращої продуктивності

```css

    transform: none !important;
  }
}
```

**JavaScript:**

```javascript
class LanguageManager {
  constructor() {
    this.currentLang = localStorage.getItem('lang') || 'en';
    this.translations = {};
    this.init();
  }

  async init() {
    // Завантажуємо переклади
    await this.loadTranslations(this.currentLang);

    // Встановлюємо поточну мову
    this.setLanguage(this.currentLang);

    // Слухаємо кліки на перемикачі
    document.querySelectorAll('[data-lang]').forEach(el => {
      el.addEventListener('click', (e) => {
        e.preventDefault();
        const lang = e.currentTarget.dataset.lang;
        this.setLanguage(lang);
      });
    });
  }

  async loadTranslations(lang) {
    try {
      const response = await fetch(`/api/translations/${lang}`);
      this.translations[lang] = await response.json();
    } catch (error) {
      console.error(`Failed to load translations for ${lang}:`, error);
      this.translations[lang] = {};
    }
  }

  setLanguage(lang) {
    this.currentLang = lang;
    localStorage.setItem('lang', lang);

    // Оновлюємо інтерфейс
    this.updateUI();

    // Оновлюємо переклади
    this.applyTranslations();
  }

  updateUI() {
    const langNames = { en: 'EN', uk: 'UA' };
    document.getElementById('currentLang').textContent = langNames[this.currentLang];

    // Оновлюємо активний елемент у випадаючому меню
    document.querySelectorAll('[data-lang]').forEach(el => {
      el.classList.toggle('active', el.dataset.lang === this.currentLang);
    });
  }

  applyTranslations() {
    const translations = this.translations[this.currentLang] || {};

    // Знаходимо всі елементи з data-i18n атрибутом
    document.querySelectorAll('[data-i18n]').forEach(el => {
      const key = el.dataset.i18n;
      if (translations[key]) {
        el.textContent = translations[key];
      }
    });

    // Оновлюємо placeholder для полів вводу
    document.querySelectorAll('[data-i18n-placeholder]').forEach(el => {
      const key = el.dataset.i18nPlaceholder;
      if (translations[key]) {
        el.placeholder = translations[key];
      }
    });
  }

  // Функція для отримання перекладу за ключом
  t(key) {
    return this.translations[this.currentLang]?.[key] || key;
  }
}

// Ініціалізація
document.addEventListener('DOMContentLoaded', () => {
  window.i18n = new LanguageManager();
});
```

## 5. Конфігурація домену та деплою

- `tuyyo-api.sashka-desire.workers.dev` — API (Cloudflare Worker)

- `api/src/index.js` — CORS origins
- `web/public/index.html` — API URL
- `web/public/booking.html` — API URL
- `web/public/admin/index.html` — API URL
